      match d with
      | Dba.Instr.Assign (_lhs, _expr, _id) -> Printf.printf "Assign\n" 
      | Dba.Instr.SJump (_addr, _tag) -> Printf.printf "Sjump\n"
      | Dba.Instr.DJump (_e_addr, _tag) -> Printf.printf "Djump\n"
      | Dba.Instr.If (_e, _addr, _int_addr) -> Printf.printf "If\n"
